# Enable NAT64 and DNS64

# Add new unmanaged interface with device nat64
# in /etc/config/network
config interface 'nat64'
        option proto 'none'
        option device 'nat64'
        option force_link '1'
        option defaultroute '0'
        option delegate '0'
# and add this to zone LAN
# in /etc/config/firewall
config zone
        option name 'lan'
        option input 'ACCEPT'
        option output 'ACCEPT'
        option forward 'ACCEPT'
        list network 'lan0'
        list network 'nat64'

# Create twor static routes: one for ipv4, one for ipv6
# in /etc/config/network
config route
        option interface 'nat64'
        option target '172.16.250.0/24'

config route6
        option interface 'nat64'
        option target '64:ff9b::/96'

# Setup port dnsmasq to non default. Eg 5353
config dnsmasq
        option port '5353'

# Setup totd to default NS server port,
# set upstream local dnsmasq server and port 5353
# and set prefix 64:ff9b::
# in /etc/config/totd
config totd
        option port '53'
        list forwarder '127.0.0.1#53'
        list prefix '64:ff9b::'

# You must run uci-default script in /rom/etc/uci-defaults/20-nat64
# for automated setup interfaces routes and firewall zone
