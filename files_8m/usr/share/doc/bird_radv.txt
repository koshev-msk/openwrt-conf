log syslog all;
router id 192.168.1.1;

ipv6 table radv6;

protocol device {
	scan time 20;
};

protocol direct {
	ipv6 { table radv6; };
	interface "br-lan";
};

protocol radv {
	ipv6 { table radv6; export all; };
	interface "br-lan" {
		max ra interval 5;
		managed no;
		prefix ::/0 {
			autonomous on;
		};
	};
};
